module pi_filter {

    port i {
        type = pasive;
    }
    
    port o {
        type = pasive;
    }
    
    port gnd {
        type = power;
    }

    resistor r1 {
        name = "R1";
        value = 100;
    }
    
    capacitor c1 {
        name = "C1";
        value = "100E-9";
    }
    
    capacitor c2 {
        name = "C2";
        value = "100E-9";
    }
    
    link i, c1.a, r1.a
    link o, c2.a, r1.b
    link gnd, c1.b, c2.b;
}
